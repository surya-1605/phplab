<?php
if (isset($_GET['choice'], $_GET['filename'])) {
    $choice = (int)$_GET['choice'];
    $file = $_GET['filename'];
    $txt = $_GET['content'] ?? "";

    switch ($choice) {
        case 1: // create
            $f = fopen($file, "w");
            if ($f) { fclose($f); echo "File Created"; }
            break;
        case 2: // write
            file_put_contents($file, $txt);
            echo "Written";
            break;
        case 3: // read
            echo file_exists($file) ? nl2br(file_get_contents($file)) : "Not Found";
            break;
        case 4: // copy
            echo copy($file, "copy_$file") ? "Copied" : "Copy Failed";
            break;
        case 5: // delete
            echo unlink($file) ? "Deleted" : "Delete Failed";
            break;
        default:
            echo "Invalid choice";
    }
}
?>

            echo "Reverse: " . strrev($in) . "<br>";
            echo "Upper: " . strtoupper($in) . "<br>";
            break;
        case 2:
            $n = (float)$in;
            echo "Absolute: " . abs($n) . "<br>";
            echo "Square root: " . sqrt(abs($n)) . "<br>";
            echo "Power (nÂ²): " . pow($n, 2);
            break;

            echo "Date: " . date("Y-m-d") . "<br>";
