<form method="get">
  <input type="radio" name="choice" value="1"> Sum of Digits<br>
  <input type="radio" name="choice" value="2"> Factorial<br>
  <input type="text" name="b" placeholder="Enter a number"><br><br>
  <input type="submit" value="Submit">
</form>

<?php
if (isset($_GET['choice'], $_GET['b'])) {
    $n = (int)$_GET['b'];
    $choice = (int)$_GET['choice'];
    switch ($choice) {
        case 1:
            $sum = array_sum(str_split((string)abs($n)));
            echo "<p>Sum of digits of $n is: $sum</p>";
            break;
        case 2:
            if ($n < 0) {
                echo "<p>Factorial not defined for negative numbers.</p>";
            } else {
                $fact = ($n === 0) ? 1 : array_product(range(1, $n));
                echo "<p>Factorial of $n is: $fact</p>";
            }
            break;
        default:
            echo "<p>Invalid choice</p>";
    }
}
?>
