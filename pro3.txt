<body>
<form method="post">
  Enter string: <input type="text" name="str" value="<?= htmlspecialchars($_POST['str'] ?? '') ?>">
  <input type="submit" value="Count">
</form>

<?php
if ($_SERVER["REQUEST_METHOD"] === "POST") {
    $str = $_POST['str'] ?? '';
    $upper = $lower = 0;
    for ($i = 0; $i < strlen($str); $i++) {
        if (ctype_upper($str[$i])) $upper++;
        elseif (ctype_lower($str[$i])) $lower++;
    }
    echo "Uppercase letters: $upper<br>";
    echo "Lowercase letters: $lower";
}
?>
